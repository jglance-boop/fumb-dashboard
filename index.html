View Payout Query 
WITH fumb_first_deposit AS (
  SELECT d.customer_token,
         MIN(d.received_at::DATE) AS first_deposit_date
  FROM app_cash.bitcoin.lightning_deposit_summary d
  JOIN app_cash.bitcoin.lnurl_receive_invoice_generation_event p
    ON d.payment_hash = p.payment_hash
  WHERE d.latest_state = 'RECEIVED'
    AND p.partner = 'FUMB'
  GROUP BY 1
),
first_buy_after_deposit AS (
  SELECT t.customer_token,
         MIN(t.latest_occurred_at::DATE) AS first_buy_date
  FROM app_cash.app.bitcoin_transactions_summary t
  JOIN fumb_first_deposit fd
    ON t.customer_token = fd.customer_token
  WHERE t.latest_state = 'SETTLED'
    AND t.revised_type = 'Buy'
    AND NOT (t.revised_type = 'Buy' AND t.subtype = 'Round Up')
    AND t.latest_occurred_at::DATE >= fd.first_deposit_date
  GROUP BY 1
)
SELECT first_buy_date AS ds,
       COUNT(*)       AS new_customers,
       SUM(COUNT(*)) OVER (ORDER BY first_buy_date) AS cumulative_customers,
       SUM(COUNT(*)) OVER (ORDER BY first_buy_date) * 2.0 AS cumulative_payout_usd
FROM first_buy_after_deposit
WHERE first_buy_date >= '2026-03-17'
  AND first_buy_date < CURRENT_DATE
GROUP BY 1
ORDER BY 1
